237template<typename T>
238T AudioBuffer<T>::getMagnitude(int channel, int startSample, int numSamples) const {
239 if (channel < 0 || channel >= m_numChannels || startSample < 0) {
240 return static_cast<T>(0);
241 }
242
243 int maxSamples = m_numSamples - startSample;
244 int samplesToDo = (std::min)(numSamples, maxSamples);
245
246 const T* data = m_data[channel].data() + startSample;
247 T maxVal = static_cast<T>(0);
248
249 for (int i = 0; i < samplesToDo; ++i) {
250 T absVal = (data[i] < static_cast<T>(0)) ? -data[i] : data[i];
251 if (absVal > maxVal) {
252 maxVal = absVal;
253 }
254 }
255
256 return maxVal;
