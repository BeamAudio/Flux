95 // Sync Tracks

100 if (reel && reel->getNodeId() == track.nodeId) { exists = true; break; }

102 if (!exists) {
103 float x = (400.0f + (track.trackIndex * 50.0f) + m_panX) / m_zoom;
104 float y = (100.0f + (track.trackIndex * 150.0f) + m_panY) / m_zoom;
105 auto reel = std::make_shared<TapeReel>(track.node, track.nodeId, x, y);
106 setupModule(reel);
107 }
108 }

111 auto nodes = m_project->getGraph()->getNodes();
112 for (auto const& [id, node] : nodes) {

114 bool exists = false;
115 for (auto& mod : m_modules) {
116 auto audioMod = std::dynamic_pointer_cast<AudioModule>(mod);
117 if (audioMod && audioMod->getNodeId() == id) { exists = true; break; }
118 }
119 if (!exists) {
120 float x = (node->getName() == "Master" ? 800.0f : 100.0f);
121 float y = (node->getName() == "Master" ? 250.0f : 100.0f);
122 auto mod = std::make_shared<AudioModule>(node, id, x, y);

272 bool onMouseUp(float x, float y, int button) override {
273 if (m_isDraggingCable) {
274 float vmx = (x - m_panX) / m_zoom;
275 float vmy = (y - m_panY) / m_zoom;
276
277 for (auto& mod : m_modules) {
278 auto audioMod = std::dynamic_pointer_cast<AudioModule>(mod);
279 if (audioMod) {
280 auto checkPort = [&](std::shared_ptr<Port> p) {
281 if (!p) return false;
282 Rect b = p->getBounds();
283 float padding = 15.0f / m_zoom; // Scale padding with zoom
284 return (vmx >= b.x - padding && vmx <= b.x + b.w + padding && vmy >= b.y - padding && vmy <= b.y + b.h + padding);
285 };
286
287 if (checkPort(audioMod->getInputPort())) connectPorts(m_activePort, audioMod->getInputPort().get());
288 else if (checkPort(audioMod->getOutputPort())) connectPorts(m_activePort, audioMod->getOutputPort().get());
289 }
290 }
291 m_isDraggingCable = false; m_activePort = nullptr;
292 return true;
293 }
294 if (!m_isVisible) return false;
295 m_isPanning = false;
296 for (auto& mod : m_modules) mod->onMouseUp(x, y, button);
297 return true;
298 }

316 bool onMouseWheel(float x, float y, float delta) override {
317 if (!m_isVisible) return false;
318 float oldZoom = m_zoom;
319 float factor = (delta > 0) ? 1.1f : 0.9f;
320 m_zoom *= factor;
321 m_zoom = (std::clamp)(m_zoom, 0.1f, 5.0f);
322 m_panX = x - (x - m_panX) * (m_zoom / oldZoom);
323 m_panY = y - (y - m_panY) * (m_zoom / oldZoom);
