218 void sliceRegion(TrackData& track, size_t index, size_t offsetInFrames) {
219 if (offsetInFrames <= 0 || offsetInFrames >= track.regions[index].duration) return;
220 Region original = track.regions[index];
221 track.regions[index].duration = offsetInFrames;
222 float ratio = (float)offsetInFrames / original.duration;
223
224 std::vector<std::vector<float>> peaks1, peaks2;
225 for (auto const& channel : original.channelPeaks) {
226 size_t peakSplit = (size_t)(channel.size() * ratio);
227 peaks1.push_back(std::vector<float>(channel.begin(), channel.begin() + peakSplit));
228 peaks2.push_back(std::vector<float>(channel.begin() + peakSplit, channel.end()));
229 }
230
231 track.regions[index].channelPeaks = peaks1;
232 Region secondHalf = {original.name + " (Slice)", original.startFrame + offsetInFrames, original.duration - offsetInFrames, original.sourceOffset + offsetInFrames, original.trackIndex, peaks2};
233 track.regions.push_back(secondHalf);
234 }
