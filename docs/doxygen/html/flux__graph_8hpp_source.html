66 std::shared_ptr<RenderPlan> compile(int bufferSizeFrames, int channels = 2) {
67 std::lock_guard<std::mutex> lock(m_mutex);
68 auto plan = std::make_shared<RenderPlan>();
69
70 // 1. Topological Sort
71 std::vector<size_t> schedule;
72 std::map<size_t, int> inDegree;
73 for (auto const& [id, node] : m_nodes) inDegree[id] = 0;
74
75 for (const auto& conn : m_connections) {
76 inDegree[conn.dstNodeId]++;
77 }
78
79 std::vector<size_t> queue;
80 for (auto const& [id, degree] : inDegree) {
81 if (degree == 0) queue.push_back(id);
82 }
83
84 while (!queue.empty()) {
85 size_t u = queue.back();
86 queue.pop_back();
87 schedule.push_back(u);
88
89 for (const auto& conn : m_connections) {
90 if (conn.srcNodeId == u) {
91 inDegree[conn.dstNodeId]--;
92 if (inDegree[conn.dstNodeId] == 0) {
93 queue.push_back(conn.dstNodeId);
94 }
95 }
96 }
97 }
98
99 // 2. Build Execution Plan
100 size_t bufSizeFloats = bufferSizeFrames * channels;
101
102 // Identify all input buffers that need clearing
103 for (const auto& [id, node] : m_nodes) {
104 for (int i = 0; i < (int)node->getInputPorts().size(); ++i) {
105 plan->clearOps.push_back({ node, i });
106 }
107 }
108
109 // Map ID to Node Ptr for quick lookup
110 std::map<size_t, std::shared_ptr<FluxNode>> nodeLookup = m_nodes;
111
112 for (size_t nodeId : schedule) {
113 auto node = nodeLookup[nodeId];
114 NodeExecution exec;
115 exec.node = node;
116
117 // Pre-calculate routing for this node's outputs
118 for (const auto& conn : m_connections) {
119 if (conn.srcNodeId == nodeId) {
120 auto dstNode = nodeLookup[conn.dstNodeId];
121 if (dstNode) {
122 exec.outgoingRoutes.push_back({
123 node,
124 conn.srcPortIdx,
125 dstNode,
126 conn.dstPortIdx
127 });
128 }
129 }
130 }
131 plan->sequence.push_back(exec);
132 }
133
134 m_needsRebuild = false;
135 return plan;
136 }
